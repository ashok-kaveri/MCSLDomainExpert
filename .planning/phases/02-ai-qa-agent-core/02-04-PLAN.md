```python
def _get_carrier_config_steps(carrier_name: str, action: str = "add") -> list[str]:
    """Returns step descriptions for carrier account configuration."""
    base_steps = [
        f"Navigate to App Settings (settings/0)",
        f"Click the Carriers tab inside the app iframe",
    ]
    if action == "add":
        base_steps += [
            f"Click Add Carrier button",
            f"Select {carrier_name} from the carrier type dropdown",
            f"Fill in the {carrier_name} account credentials",
            f"Click Save to complete the configuration",
            f"Verify {carrier_name} appears in the carriers list",
        ]
    elif action == "edit":
        base_steps += [
            f"Find the {carrier_name} row in the carriers list",
            f"Click the Edit button for {carrier_name}",
            f"Update the credentials as required",
            f"Click Save",
        ]
    return base_steps
```
